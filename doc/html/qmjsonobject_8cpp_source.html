61 void QMJsonObject::clear(void)

107 void QMJsonObject::insert(const QString &key, const QMPointer<QMJsonValue> &value, QMJsonReplacementPolicy policy)
108 {
109  if(key.isEmpty() == true)
110  return;
111 
112  auto iter = mHash.find(key);
113 
114  if(iter != mHash.end())
115  {
116  switch(policy)
117  {
118  case QMJsonReplacementPolicy_Replace:
119  this->erase(iter);
120  break;
121 
122  case QMJsonReplacementPolicy_Ignore:
123  return;
124  };
125  }
126 
127  if(value.isNull() == true)
128  {
129  auto newValue = QMPointer<QMJsonValue>(new QMJsonValue);
130 
131  mHash.insert(key, newValue);
132  emit itemAdded(key, newValue);
133  }
134  else
135  {
136  mHash.insert(key, value);
137  emit itemAdded(key, value);
138  }
139 }

141 void QMJsonObject::unite(const QMPointer<QMJsonObject> &object, QMJsonReplacementPolicy replacementPolicy, QMJsonArrayUnitePolicy unitePolicy)
142 {
143  if(object.isNull() == true)
144  return;
145 
146  for(const auto &key : object->keys())
147  {
148  if(mHash.contains(key) == true)
149  {
150  const auto &value1 = this->value(key);
151  const auto &value2 = object->value(key);
152 
153  if(value1->isObject() && value2->isObject())
154  {
155  value1->toObject()->unite(value2->toObject(), replacementPolicy, unitePolicy);
156  continue;
157  }
158 
159  if(value1->isArray() && value2->isArray())
160  {
161  value1->toArray()->unite(value2->toArray(), unitePolicy);
162  continue;
163  }
164 
165  this->insert(key, value2, replacementPolicy);
166  }
167  else
168  {
169  this->insert(key, object->value(key), replacementPolicy);
170  }
171  }
172 }

238 const QHash<QString, QMPointer<QMJsonValue> >::iterator QMJsonObject::erase(const QHash<QString, QMPointer<QMJsonValue> >::iterator &iter)
239 {
240  if(iter == mHash.end())
241  return mHash.end();
242 
243  auto key = iter.key();
244  auto value = iter.value();
245 
246  auto niter = mHash.erase(iter);
247  emit itemRemoved(key, value);
248 
249  return niter;
