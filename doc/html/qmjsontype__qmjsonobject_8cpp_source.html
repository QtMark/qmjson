98 
99  switch(sort)
100  {
101  case QMJsonSort_None:
102  keys = object->keys();
103  break;
104 
105  case QMJsonSort_CaseInsensitive:
106  keys = object->keys();
107  keys.sort(Qt::CaseInsensitive);
108  break;
109 
110  case QMJsonSort_CaseSensitive:
111  keys = object->keys();
112  keys.sort(Qt::CaseSensitive);
113  break;
114  }
115 
116  if(tab == (int32_t)QMJsonFormat_Optimized)
117  {
118  json += '{';
119 
120  for(const auto &key : keys)
121  {
122  const auto &value = object->value(key);
123 
124  json += '"';
125  json += key;
126  json += "\":";
127  json += value->toJson((QMJsonFormat)tab, sort);
128  json += ',';
129  }
130 
131  return json.replace(json.length() - 1, 1, '}');
132  }
133  else
134  {
135  json += '{';
136 
137  tab += 4;

139  for(const auto &key : keys)
140  {

142 
143  json += "\r\n";
144  json += space;

147  json += "\":";
148  json += value->toJson((QMJsonFormat)tab, sort);
149  json += ',';
150 
151  }
